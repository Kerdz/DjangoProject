pipeline {
    agent any

    environment {
        EC2_USER = "ubuntu"
        EC2_HOST = "18.222.119.100"
        PROJECT_DIR = "/home/ubuntu/DjangoProject"
    }

    triggers {
        githubPush()
    }

    stages {
        stage('Update Code on EC2') {
            steps {
                script {
                    sshagent (credentials: ['ec2-ssh-private-key']) {
                        sh """
                        ssh -o StrictHostKeyChecking=no ${EC2_USER}@${EC2_HOST} << 'ENDSSH'
                        cd ${PROJECT_DIR}
                        git pull origin main
                        
                        # Set up virtual environment if not exists
                        if [ ! -d "myEnv" ]; then
                            python3 -m venv myEnv
                        fi
                        
                        source myEnv/bin/activate
                        pip install --upgrade pip
                        pip install -r requirements.txt
                        ENDSSH
                        """
                    }
                }
            }
        }
    }

    post {
        success {
            echo "Code updated and dependencies installed successfully on EC2!"
        }
        failure {
            echo "Deployment failed. Check Jenkins logs and EC2 connection."
        }
    }
}
